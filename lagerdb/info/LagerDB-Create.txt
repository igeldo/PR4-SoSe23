CREATE TABLE LAGER (LNR INT, ORT CHAR(20), LCODE CHAR(1), MENGE INT, PRIMARY KEY(LNR));CREATE TABLE ROHSTOFF (RNR INT, RNAME CHAR(20), RCODE CHAR(1), GEBINDE CHAR(20), PRIMARY KEY(RNR));CREATE TABLE PRODUKT (PNR INT, PNAME CHAR(20), PREIS INT, PRIMARY KEY(PNR));CREATE TABLE LR (LNR INT, RNR INT, MENGE INT, PRIMARY KEY(LNR,RNR), FOREIGN KEY (LNR) REFERENCES LAGER, FOREIGN KEY (RNR) REFERENCES ROHSTOFF);CREATE TABLE PR (PNR INT, RNR INT, MENGE INT, PRIMARY KEY(PNR,RNR), FOREIGN KEY (PNR) REFERENCES PRODUKT, FOREIGN KEY (RNR) REFERENCES ROHSTOFF);
